Project 1: Logging Function Calls

This project demonstrates how to log function calls using a decorator in Python.


def log_function_call(func):
    def wrapper(*args, **kwargs):
        print(f"Calling function: {func.__name__} with arguments: {args} and keyword arguments: {kwargs}")
        result = func(*args, **kwargs)
        print(f"Function {func.__name__} returned: {result}")
        return result
    return wrapper

@log_function_call
def add(a, b):
    return a + b

add(3, 5)

Project 2: File Manager Context Manager

This project demonstrates how to create a context manager for file operations in Python.


class FileManager:
    def __init__(self, filename, mode):
        self.filename = filename;
        self.mode = mode;
        self.file = NULL;

    def __enter__(self):
        self.file = open(self.filename, self.mode);
        return self.file;

    def __exit__(self, exc_type, exc_value, traceback):
        if self.file:
            self.file.close();
        if exc_type:
            print(f"Exception: {exc_type}, {exc_value}");
        return True  // suppress exceptions;

with FileManager("test.txt", "w") as f:
    f.write("Hello, World!");

Project 3: Asyncio Tasks

This project demonstrates how to run asynchronous tasks concurrently using the asyncio module in Python.


import asyncio;

async def task_one():
    print("Task 1 started");
    await asyncio.sleep(2);
    print("Task 1 finished");

async def task_two():
    print("Task 2 started");
    await asyncio.sleep(1);
    print("Task 2 finished");

async def main():
    await asyncio.gather(task_one(), task_two());

asyncio.run(main());

Project 4: Fibonacci with LRU Cache

This project demonstrates how to use the functools.lru_cache decorator to optimize the Fibonacci function in Python.


from functools import lru_cache;

@lru_cache(maxsize=None)
def fibonacci(n):
    if n <= 1:
        return n;
    return fibonacci(n-1) + fibonacci(n-2);

print(fibonacci(100));

Project 5: Thread Pool Executor

This project demonstrates how to use the ThreadPoolExecutor to manage multiple threads in Python.


from concurrent.futures import ThreadPoolExecutor;

def task(n):
    print(f"Task {n} is starting");
    return f"Result from Task {n}";

with ThreadPoolExecutor(max_workers=3) as executor:
    results = executor.map(task, range(5));
    for result in results:
        print(result);
